# periodic_monitor.sh yapılandırma dosyası şablonu
# Bu dosyayı Agent-shared/periodic_monitor_config.txt olarak kopyalayıp kullanın
# PM/SOLO aracıları gerekirse değerleri ayarlayabilir

# Üstüne yazma güncelleme sıklığı (saniye)
# Varsayılan: 30 saniye
# Önerilen aralık: 10-300 saniye
UPDATE_INTERVAL_SEC=30

# Kilometre taşı kontrol aralığı (dakika)
# Varsayılan: 30 dakika
# Bu aralıkta kilometre taşlarını (30,60,90,120,180 dk) kontrol eder
MILESTONE_INTERVAL_MIN=30

# Azami çalışma süresi (dakika)
# Varsayılan: 4320 dakika (3 gün)
# Projenin öngörülen süresine göre ayarlayın
MAX_RUNTIME_MIN=4320

# Bütçe toplamı yürütme aralığı (dakika)
# Varsayılan: 3 dakika
# Önerilen aralık: 1-10 dakika
BUDGET_INTERVAL_MIN=3

# MILESTONE_INTERVAL_MIN=5
# MAX_RUNTIME_MIN=60
